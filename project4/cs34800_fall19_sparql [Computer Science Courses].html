Query 1 (Triple Pattern): Get the birth date of Muhammad Ali 

PREFIX dbo: <http://dbpedia.org/ontology/>
SELECT  ?bd 
WHERE {
:Muhammad_Ali dbo:birthDate ?bd .
}

Query 2 (Multiple Triple Patterns): Get names of all Boxers 

PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX dbo: <http://dbpedia.org/ontology/>
SELECT ?name
WHERE {
  ?uri rdf:type    dbo:Boxer. 
  ?uri rdfs:label  ?name . 
}

Query 3 (Filter): Get names of all Boxers in English 

PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX dbo: <http://dbpedia.org/ontology/>
SELECT ?name
WHERE {
  ?uri rdf:type    dbo:Boxer. 
  ?uri rdfs:label  ?name . 
  FILTER ( lang(?name) = 'en')
}

Query 4 (OPTIONAL): Get names of all Boxers and show nicknames if exists 

PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX dbo: <http://dbpedia.org/ontology/>
PREFIX foaf: <http://xmlns.com/foaf/0.1/>

SELECT ?resource ?label ?nickname
WHERE {
  ?resource rdf:type dbo:Boxer .
  ?resource rdfs:label ?lbl .
  OPTIONAL { ?resource foaf:nick ?nickname . }
  FILTER(lang(?lbl) = 'en')
}

Query 5 (MINUS): Get names of all Boxers that do not have a nickname 

PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX dbo: <http://dbpedia.org/ontology/>
PREFIX foaf: <http://xmlns.com/foaf/0.1/>



SELECT ?label
WHERE {
  ?resource rdf:type dbo:Boxer .
  ?resource rdfs:label ?lbl .
  MINUS { ?resource foaf:nick ?nickname . }
}

Query 6 (Property Paths – Alternative Paths ( | )): Get name or titles of Muhammad Ali 

PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX dbpedia2: <http://dbpedia.org/property/>
PREFIX : <http://dbpedia.org/resource/>

SELECT ?var
WHERE {
   :Muhammad_Ali (dbp:title | rdfs:label) ?var .
}

Query 7 (Property Paths – Using Regular Expression): Get all heavy weight champions before Muhammad Ali 

PREFIX dbpedia2: <http://dbpedia.org/property/>

SELECT ?champions
WHERE {
   ?champions dbp:before+ :Muhammad_Ali .
}

Query 8 (Property Paths – Using Defined Paths): Get all heavy weight champions before Muhammad Ali that are two links away 

PREFIX dbpedia2: <http://dbpedia.org/property/>
SELECT ?s
WHERE {
  :Muhammad_Ali dbp:before/dbp:before ?s .
}

Query 9 (Property Paths – Negation): Get all heavy weight champions that Muhammad Ali is not before them 

PREFIX dbpedia2: <http://dbpedia.org/property/>

SELECT ?champions
WHERE {
   :Muhammad_Ali ^dbp:before ?champions .
}

Query10 (DISTINCT - Eliminating Redundant Output): Get all unique predicates/relations for the Muhammed Ali 

PREFIX : <http://dbpedia.org/resource/>

SELECT DISTINCT ?predicate
WHERE {
   :Muhammad_Ali ?predicate ?o .
}

Query 11 (UNION): Get the champion before and after Muhammed Ali 


PREFIX dbp: <http://dbpedia.org/property/>
PREFIX : <http://dbpedia.org/resource/>

SELECT ?champion
WHERE {
   {?champion dbp:before :Muhammad_Ali .}
   UNION
   {?champion dbp:after  :Muhammad_Ali .}
}

Query 12 (FILTER on Condition - regexp): Get matches of Muhammed Ali that contain the word “Undisputed” 

PREFIX dbp: <http://dbpedia.org/property/>
PREFIX : <http://dbpedia.org/resource/>

SELECT ?title
WHERE {
   :Muhammad_Ali dbp:title ?title .
   FILTER(regex(?title, 'Undisputed', 'i'))
   FILTER(!(isURI(?title)))

}

Query 13 (FILTER on Condition - isURI): Get matches of Muhammed Ali that contain the word “Undisputed” and is not a URI 

PREFIX dbp: <http://dbpedia.org/property/>
PREFIX : <http://dbpedia.org/resource/>

SELECT ?title
WHERE {
   :Muhammad_Ali dbp:title ?title .
   FILTER(regex(?title, 'Undisputed', 'i')
   FILTER(!(isURI(?title)))
}

Query 14 (LIMIT and OFFSET): Get two titles after the second returned title of Muhammed Ali 

PREFIX dbp: <http://dbpedia.org/property/>
PREFIX : <http://dbpedia.org/resource/>

SELECT ?title
WHERE {
   :Muhammad_Ali dbp:title ?title .
    FILTER(!(isURI(?title)))
}
OFFSET 1
LIMIT 2

Query 15 (ORDER BY – Sorting Results): Get all sorted titles of Muhammed Ali 

PREFIX dbp: <http://dbpedia.org/property/>
PREFIX : <http://dbpedia.org/resource/>

SELECT ?title
WHERE {
   :Muhammad_Ali dbp:title ?title .
}
ORDER BY(?title)
